# Insider Cluster Watch - Trading System Crontab
# Install with: crontab scripts/crontab.txt
#
# All times in UTC. ET = UTC - 5 (EST) or UTC - 4 (EDT)
# Current: EDT (UTC-4), so 9:36 AM ET = 13:36 UTC, 4:30 PM ET = 20:30 UTC
# Winter: EST (UTC-5), so 9:36 AM ET = 14:36 UTC, 4:30 PM ET = 21:30 UTC
#
# Using EST (UTC-5) schedule to be safe year-round:

HOME=/home/user

# Git pull + Morning execution: 9:36 AM ET = 14:36 UTC (weekdays only)
36 14 * * 1-5 $HOME/insider-cluster-watch/scripts/run_git_pull.sh >> $HOME/insider-cluster-watch/logs/git_pull_$(date +\%Y\%m\%d).log 2>&1 && $HOME/insider-cluster-watch/scripts/run_morning.sh >> $HOME/insider-cluster-watch/logs/morning_$(date +\%Y\%m\%d).log 2>&1

# Position monitoring: Every 5 minutes during market hours (14:35-21:00 UTC)
35-59/5 14 * * 1-5 $HOME/insider-cluster-watch/scripts/run_monitor.sh >> $HOME/insider-cluster-watch/logs/monitor_$(date +\%Y\%m\%d).log 2>&1
*/5 15-20 * * 1-5 $HOME/insider-cluster-watch/scripts/run_monitor.sh >> $HOME/insider-cluster-watch/logs/monitor_$(date +\%Y\%m\%d).log 2>&1
0 21 * * 1-5 $HOME/insider-cluster-watch/scripts/run_monitor.sh >> $HOME/insider-cluster-watch/logs/monitor_$(date +\%Y\%m\%d).log 2>&1

# EOD summary: 4:30 PM ET = 21:30 UTC
30 21 * * 1-5 $HOME/insider-cluster-watch/scripts/run_eod.sh >> $HOME/insider-cluster-watch/logs/eod_$(date +\%Y\%m\%d).log 2>&1

# Health check: Every 6 hours
0 */6 * * * $HOME/insider-cluster-watch/scripts/health_check.sh
